```python
"""KalshiAdapter: canonical wrapper over KalshiClient (transport tier)."""
from __future__ import annotations
import httpx
from datetime import datetime, timezone
from typing import TYPE_CHECKING

from sharpedge_venue_adapters.protocol import (
    VenueCapability,
    CanonicalMarket,
    CanonicalOrderBook,
    CanonicalTrade,
    VenueFeeSchedule,
    SettlementState,
    MarketStatePacket,
    MarketLifecycleState,
)

KALSHI_API_BASE = "https://trading-api.kalshi.com/trade-api/v2"
KALSHI_TAKER_FEE = 0.07


class KalshiAdapter:
    venue_id: str = "kalshi"
    capabilities: VenueCapability = VenueCapability(
        read_only=False,
        streaming_quotes=False,
        streaming_orderbook=False,
        execution_supported=True,
        maker_rewards=False,
        settlement_feed=True,
    )

    def __init__(self, api_key: str | None = None) -> None:
        self._api_key = api_key
        try:
            from sharpedge_feeds.kalshi_client import KalshiClient
            self._client = KalshiClient(api_key=api_key) if api_key else None
        except ImportError:
            self._client = None

    async def list_markets(self, status: str = "open") -> list[CanonicalMarket]:
        if self._client is None:
            return []
        try:
            raw = await self._client.get_markets()
            return [self._to_canonical(m) for m in raw]
        except Exception:
            return []

    def _to_canonical(self, m) -> CanonicalMarket:
        state = MarketLifecycleState.SETTLED if m.result in ("yes", "no") else MarketLifecycleState.OPEN
        return CanonicalMarket(
            venue_id="kalshi",
            market_id=m.ticker,
            title=m.title,
            state=state,
            yes_bid=float(m.yes_bid) if m.yes_bid is not None else 0.0,
            yes_ask=float(m.yes_ask) if m.yes_ask is not None else 0.0,
            volume=int(m.volume) if m.volume is not None else 0,
            close_time_utc=str(m.close_time) if m.close_time else "",
        )

    async def get_market_details(self, market_id: str) -> CanonicalMarket | None:
        markets = await self.list_markets()
        return next((m for m in markets if m.market_id == market_id), None)

    async def get_orderbook(self, market_id: str) -> CanonicalOrderBook:
        if self._api_key is None:
            return CanonicalOrderBook(bids=[], asks=[], timestamp_utc=datetime.now(timezone.utc).isoformat())
        try:
            async with httpx.AsyncClient() as client:
                resp = await client.get(f"{KALSHI_API_BASE}/markets/{market_id}/orderbook")
                resp.raise_for_status()
                data = resp.json()
                return CanonicalOrderBook(
                    bids=data.get("orderbook", {}).get("yes", []),
                    asks=data.get("orderbook", {}).get("no", []),
                    timestamp_utc=datetime.now(timezone.utc).isoformat(),
                )
        except Exception:
            return CanonicalOrderBook(bids=[], asks=[], timestamp_utc=datetime.now(timezone.utc).isoformat())

    async def get_trades(self, market_id: str, since=None) -> list[CanonicalTrade]:
        return []  # Phase 6: not yet fetching live trades

    async def get_historical_snapshots(self, market_id: str) -> list[MarketStatePacket]:
        raise NotImplementedError(
            "historical data endpoint not confirmed — implement after verifying Kalshi candlestick API"
        )

    async def get_fees_and_limits(self) -> VenueFeeSchedule:
        return VenueFeeSchedule(
            venue_id="kalshi",
            maker_fee_rate=0.0,
            taker_fee_rate=KALSHI_TAKER_FEE,
            expected_quote_refresh_seconds=5,
        )

    async def get_settlement_state(self, market_id: str) -> SettlementState | None:
        market = await self.get_market_details(market_id)
        if market is None:
            return None
        return SettlementState(
            market_id=market_id,
            outcome=None,  # TODO: map from market.result when live client available
            is_settled=market.state == MarketLifecycleState.SETTLED,
        )
```

```python
"""PolymarketAdapter: canonical wrapper over PolymarketClient (transport tier)."""
from __future__ import annotations
import httpx
from datetime import datetime, timezone

from sharpedge_venue_adapters.protocol import (
    VenueCapability,
    CanonicalMarket,
    CanonicalOrderBook,
    CanonicalTrade,
    VenueFeeSchedule,
    SettlementState,
    MarketStatePacket,
    MarketLifecycleState,
)

POLYMARKET_CLOB_BASE = "https://clob.polymarket.com"


class PolymarketAdapter:
    venue_id: str = "polymarket"
    capabilities: VenueCapability = VenueCapability(
        read_only=True,
        streaming_quotes=False,
        streaming_orderbook=False,
        execution_supported=False,
        maker_rewards=True,
        settlement_feed=False,
    )

    def __init__(self) -> None:
        try:
            from sharpedge_feeds.polymarket_client import PolymarketClient
            self._client = PolymarketClient()
        except ImportError:
            self._client = None

    async def list_markets(self, status: str = "open") -> list[CanonicalMarket]:
        if self._client is None:
            return []
        try:
            raw = await self._client.get_markets()
            return [self._to_canonical(m) for m in raw if m is not None]
        except Exception:
            return []

    def _to_canonical(self, m) -> CanonicalMarket:
        # Binary market: use YES outcome price as yes_bid, infer yes_ask
        yes_price = 0.5
        for outcome in getattr(m, "outcomes", []):
            if str(getattr(outcome, "outcome", "")).upper() == "YES":
                yes_price = float(getattr(outcome, "price", 0.5))
                break
        return CanonicalMarket(
            venue_id="polymarket",
            market_id=getattr(m, "condition_id", ""),
            title=getattr(m, "question", ""),
            state=MarketLifecycleState.OPEN,
            yes_bid=yes_price - 0.01,
            yes_ask=yes_price + 0.01,
            volume=int(getattr(m, "volume", 0) or 0),
            close_time_utc="",
        )

    async def get_market_details(self, market_id: str) -> CanonicalMarket | None:
        markets = await self.list_markets()
        return next((m for m in markets if m.market_id == market_id), None)

    async def get_orderbook(self, market_id: str) -> CanonicalOrderBook:
        # market_id here is a token_id for CLOB orderbook queries
        try:
            async with httpx.AsyncClient() as client:
                resp = await client.get(f"{POLYMARKET_CLOB_BASE}/book", params={"token_id": market_id})
                resp.raise_for_status()
                data = resp.json()
                return CanonicalOrderBook(
                    bids=data.get("bids", []),
                    asks=data.get("asks", []),
                    timestamp_utc=datetime.now(timezone.utc).isoformat(),
                )
        except Exception:
            return CanonicalOrderBook(bids=[], asks=[], timestamp_utc=datetime.now(timezone.utc).isoformat())

    async def get_trades(self, market_id: str, since=None) -> list[CanonicalTrade]:
        return []

    async def get_historical_snapshots(self, market_id: str) -> list[MarketStatePacket]:
        return []

    async def get_fees_and_limits(self) -> VenueFeeSchedule:
        return VenueFeeSchedule(
            venue_id="polymarket",
            maker_fee_rate=0.0,
            taker_fee_rate=0.0,
            expected_quote_refresh_seconds=30,
        )

    async def get_settlement_state(self, market_id: str) -> SettlementState | None:
        return None  # Polymarket settlement via Gamma API — Phase 6 read-only
```
