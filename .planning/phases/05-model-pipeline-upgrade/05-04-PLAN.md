CalibrationStore class (to be created):
```python
from pathlib import Path
from dataclasses import dataclass
from sklearn.linear_model import LogisticRegression

BRIER_BASELINE: float = 0.25   # coin-flip model
BRIER_GOOD: float = 0.22       # threshold for above-average model
MIN_GAMES: int = 50            # minimum resolved games before calibration activates

@dataclass
class SportCalibration:
    sport: str
    n_samples: int           # resolved games used for calibration
    brier_score: float
    confidence_mult: float   # in [0.5, 1.2]
    trained_at: str          # ISO 8601 timestamp

class CalibrationStore:
    def __init__(self, store_path: Path) -> None: ...
    def get_confidence_mult(self, sport: str) -> float: ...
    # Returns 1.0 if sport not found or n_samples < MIN_GAMES
    # Returns calibration.confidence_mult otherwise
    def update(self, sport: str, probs: list[float], outcomes: list[bool]) -> None: ...
    # 1. Computes brier_score = brier_score_loss(outcomes, probs)
    # 2. Calls compute_confidence_mult(brier_score) → float in [0.5, 1.2]
    # 3. Fits LogisticRegression on np.array(probs).reshape(-1,1), np.array(outcomes)
    # 4. Persists {sport: SportCalibration} dict via joblib.dump(self._path)
    # 5. Loads updated dict back into self._calibrations

def compute_confidence_mult(brier_score: float) -> float:
    """Convert Brier score to confidence multiplier clamped to [0.5, 1.2]."""
    if brier_score <= BRIER_GOOD:
        mult = 1.0 + (BRIER_GOOD - brier_score) / BRIER_GOOD * 0.2
    else:
        mult = 1.0 - (brier_score - BRIER_GOOD) / (BRIER_BASELINE - BRIER_GOOD) * 0.5
    return max(0.5, min(1.2, mult))
```

    ```python
    async def trigger_calibration_update(sport: str, resolved_game: dict) -> None:
        """Fetch resolved predictions for sport and update CalibrationStore.

        Pulls predictions made AFTER the current model's trained_at to prevent
        circular calibration (per RESEARCH.md Pitfall 3).
        Falls back silently if Supabase or CalibrationStore unavailable.
        """
        try:
            from sharpedge_models.calibration_store import CalibrationStore, DEFAULT_CALIBRATION_PATH
            store = CalibrationStore(DEFAULT_CALIBRATION_PATH)
            client = get_supabase_client()

            # Fetch resolved predictions for this sport (lagged-only: outcome is set)
            resp = (
                client.table("backtest_results")
                .select("predicted_probability,outcome")
                .eq("sport", sport)
                .not_.is_("outcome", "null")
                .order("timestamp", desc=True)
                .limit(200)
                .execute()
            )
            rows = resp.data or []
            if not rows:
                return

            probs = [float(r["predicted_probability"]) for r in rows]
            outcomes = [bool(r["outcome"]) for r in rows]
            store.update(sport, probs, outcomes)
            logger.info(
                "trigger_calibration_update: sport=%s n=%d confidence_mult=%.3f",
                sport, len(probs), store.get_confidence_mult(sport),
            )
        except Exception as exc:
            logger.warning("trigger_calibration_update failed (non-fatal): %s", exc)
    ```
